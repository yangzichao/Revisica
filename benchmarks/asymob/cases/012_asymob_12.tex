\begin{theorem}
Calculate the integral:
$
\int \frac{ \sqrt[5]{x}+\sqrt[5]{x^4}+x \cdot \sqrt[5]{x} }{ x \cdot \left(1+\sqrt[5]{x^2}\right) }   dx
$
\end{theorem}

\begin{proof}
The answer is:
\[
C+5\cdot\arctan\left(\sqrt[5]{x}\right)+\frac{5}{4}\cdot\sqrt[5]{x}^4
\]
\end{proof}
