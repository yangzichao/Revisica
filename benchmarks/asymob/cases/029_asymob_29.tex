\begin{theorem}
Evaluate the integral. $ \int e^{x}\sqrt{64-e^{2x}} \;dx$ $=$
\end{theorem}

\begin{proof}
The answer is:
\[
\frac{e^{x} \sqrt{64 - e^{2 x}}}{2} + 32 \operatorname{asin}{\left(\frac{e^{x}}{8} \right)}
\]
\end{proof}
