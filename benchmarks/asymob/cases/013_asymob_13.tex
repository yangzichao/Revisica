\begin{theorem}
Solve the integral:
$
\int \frac{ 1 }{ \sin(x)^7 \cdot \cos(x) }   dx
$
\end{theorem}

\begin{proof}
The answer is:
\[
C+\ln\left(\left|\tan(x)\right|\right)-\frac{3}{2\cdot\left(\tan(x)\right)^2}-\frac{3}{4\cdot\left(\tan(x)\right)^4}-\frac{1}{6\cdot\left(\tan(x)\right)^6}
\]
\end{proof}
