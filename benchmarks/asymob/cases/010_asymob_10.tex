\begin{theorem}
Evaluate
$ \lim_{x \to 0}\left( \left| \frac{ -\sin(x) }{ x } \right| \right)^{\frac{ 1 }{ 4 \cdot x^2 }} $
\end{theorem}

\begin{proof}
The answer is:
\[
e^{\frac{-1}{24}}
\]
\end{proof}
