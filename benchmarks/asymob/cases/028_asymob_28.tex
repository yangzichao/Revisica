\begin{theorem}
Evaluate the integral.
$ \int \frac{10x^2-48x-38}{x^3-5x^2-8x+48} dx $
\end{theorem}

\begin{proof}
The answer is:
\[
\frac{2 \left(\left(x - 4\right) \left(3 \log{\left(\left|{x - 4}\right| \right)} + 2 \log{\left(\left|{x + 3}\right| \right)}\right) + 5\right)}{x - 4}
\]
\end{proof}
