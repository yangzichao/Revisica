\begin{theorem}
Evaluate $\lim_{x \to 0^{+}}\left(\left(\frac{ \tan\left(\frac{ x }{ 2 }\right) }{ \frac{ x }{ 2 } }\right)^{\frac{ 3 }{ x^2 }}\right)$
\end{theorem}

\begin{proof}
The answer is:
\[
$e^{\frac{1}{4}}$
\]
\end{proof}
