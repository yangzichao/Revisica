\newtheorem{theorem}{Claim}

\begin{document}

\begin{theorem}
Use factoring to calculate the following limit.
$ \lim_{x \rightarrow K} \frac {{x}^4-K^4} {{x}^5-K^5} $
\end{theorem}

\begin{proof}
The answer is:
\[
\frac{4}{5 K}
\]
\end{proof}
