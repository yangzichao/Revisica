\begin{theorem}
Compute the integral:
$
-2 \cdot \int x^{-4} \cdot \left(4+x^2\right)^{\frac{ 1 }{ 2 }}   dx
$
\end{theorem}

\begin{proof}
The answer is:
\[
C+\frac{1}{6}\cdot\left(\frac{4}{x^2}+1\right)\cdot\sqrt{\frac{4}{x^2}+1}
\]
\end{proof}
