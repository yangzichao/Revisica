\begin{theorem}
Compute the integral: $ \int \frac{ -1 }{ x^2 \cdot \left(3+x^3\right)^{\frac{ 5 }{ 3 }} }   dx $
\end{theorem}

\begin{proof}
The answer is:
\[
\frac{1}{9}\cdot\sqrt[3]{1+\frac{3}{x^3}}+\frac{1}{18\cdot\left(1+\frac{3}{x^3}\right)^{\frac{2}{3}}}
\]
\end{proof}
