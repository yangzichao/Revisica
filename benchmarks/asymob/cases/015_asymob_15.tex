\begin{theorem}
Solve the integral:
$
\int \left(\frac{ x+4 }{ x-4 } \right)^{\frac{ 3 }{ 2 }}   dx
$
\end{theorem}

\begin{proof}
The answer is:
\[
C+\sqrt{\frac{x+4}{x-4}}\cdot(x-20)-12\cdot\ln\left(\left|\frac{\sqrt{x-4}-\sqrt{x+4}}{\sqrt{x-4}+\sqrt{x+4}}\right|\right)
\]
\end{proof}
