\begin{theorem}
We have
\[
\int_0^1 x^2 \, dx = \frac{1}{2}.
\]
\end{theorem}

\begin{proof}
Clearly, the antiderivative of $x^2$ is $x^2/2$.
Therefore
\[
\int_0^1 x^2 \, dx = \frac{1}{2}.
\]
Thus the theorem follows.
\end{proof}
